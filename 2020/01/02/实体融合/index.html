统计词频

TF-IDF求解函数

TF意思是词频(Term Frequency)，IDF意思是逆文本频率指数(Inverse Document Frequency)。

在求TF的时候，使用了一个子方法_inc()，是统计词频的中调用的子方法

xxxxxxxxxx
def _inc(i,dic):
    #i:a key in dic
    #dic:dic[i]++
    if i in dic: 
        dic[i]+=1
    else: 
        dic[i]=1
        
def _tf(tokens):
    tfs={}
    for i in tokens: 
        _inc(i,tfs)            
    n=float(len(tokens))
    for i in tfs:
        tfs[i]/=n
    return tfs
​
def _idf(rec2tok):
    idfs={}
    N=float(len(amazon_df)+len(google_df))
    for i in rec2tok:
        s=set(rec2tok[i])
        for j in s:
            _inc(j,idfs)
    for i in idfs:
        idfs[i]=N/idfs[i]
    return idfs
​
def tfidf(tokens,idfs):
    ans=_tf(tokens)
    s=set(tokens)
    for i in ans:
        ans[i]*= idfs[i]
    return ans

def _norm(a):
    ans=0

求余弦相似度

def buildSim(Id,weight,norm,weights,norms,inv,sims):
    #weights : Id->token->weight
    #norms : Id->norm
    for i in weight:
        if i in inv:
            for j in inv[i]:
                if not (j,Id) in sims: 
                    sims[(j,Id)]= _dotprod(weight, weights[j]) / (norm *norms[j])

    true_dup_sims=[]

            true_dup_sims.append(i)

找出高于门槛相似度的map中不在正确map中的数量

def _falsepos(threshold):
    ans=0
    for i in true_dup_sims:
        if not i in perfectMap: 
            ans+=1
    return ans

根据预测中正确的和预测的总的数量求准确率

def precision(threshold):
    _truepos(threshold)
    a=len(true_dup_sims)-_falsepos(threshold)
    b=len(true_dup_sims)
    return a*1.0/b
